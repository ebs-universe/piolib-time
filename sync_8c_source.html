105 uint16_t tm_sync_host_read_hook(uint16_t address){
106  if (tm_sync_sm.state == TM_SYNC_STATE_WAIT_DELAY_OUT){
107  tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_IN;

115 void tm_sync_handler(uint16_t addr){
116  switch(tm_sync_sm.state){
117  case TM_SYNC_STATE_PREINIT:
118  break;
119  case TM_SYNC_STATE_IDLE:
120  if (ucdm_register[addr].data){
121  // Got sync with timestamp from the host.
122  tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_OUT;
123  memcpy(
124  &tm_sync_sm.t1,
125  (void *)&ucdm_register[addr - (sizeof(tm_system_t) / 2) + 1],
126  sizeof(tm_system_t)
127  );
128  } else {
129  // Got sync without timestamp from the host.
130  tm_sync_sm.state = TM_SYNC_STATE_WAIT_FOLLOW_UP;
131  }
132  tm_current_time(&(tm_sync_sm.t1p));
133  break;
134  case TM_SYNC_STATE_WAIT_FOLLOW_UP:
135  // Got the sync timestamp from the host.
136  tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_OUT;
137  memcpy(
138  &tm_sync_sm.t1,
139  (void *)&ucdm_register[addr - (sizeof(tm_system_t) / 2) + 1],
140  sizeof(tm_system_t)
141  );
142  break;
143  case TM_SYNC_STATE_WAIT_DELAY_OUT:
144  // This is handled in the register read function for the
145  // system time register of UCDM. t2 will be set there.
146  break;
147  case TM_SYNC_STATE_WAIT_DELAY_IN:
148  // Host returned its timestamp for delay calculation.
149  memcpy(
150  &tm_sync_sm.t2p,
151  (void *)&ucdm_register[addr - (sizeof(tm_system_t) / 2) + 1],
152  sizeof(tm_system_t)
153  );
154  tm_sync_sm.state = TM_SYNC_STATE_IDLE;
155  // All information is now available. Calculate and apply.
156  tm_sync_apply();
157  ucdm_exception_status &= ~UCDM_EXST_TIMESYNC_REQ;
158  break;
159  default:
160  break;
161  }
162  return;
163 }
