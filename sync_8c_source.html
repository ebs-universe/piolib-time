58ucdm_addr_t tm_sync_init(ucdm_addr_t ucdm_address){
59 // Setup System Time Read Hook
60 ucdm_redirect_regr_func(ucdm_address, &tm_sync_host_read_hook);
61 ucdm_address += 2;
62
63 ucdm_addr_sync_host_ts = ucdm_address;
64 // Setup Host Interface Registers
65 for(uint8_t i=0; i<(sizeof(tm_system_t)/2); i++, ucdm_address++){
66 ucdm_enable_regw(ucdm_address + i);
67 }
68
69 // Setup Host Sync Handler(s)

71 &tm_avlt_sync_handler_node,
72 &tm_sync_handler);
73 tm_sync_sm.state = TM_SYNC_STATE_IDLE;
74 return ucdm_address;
75}

85static inline void tm_sync_apply(void){
86 tm_sdelta_t offset, tsd1, tsd2;
87
88 tm_get_sdelta(&(tm_sync_sm.t1), &(tm_sync_sm.t1p), &(tsd1));
89 tm_get_sdelta(&(tm_sync_sm.t2), &(tm_sync_sm.t2p), &(tsd2));
90
91 offset = (tsd1 - tsd2)/ 2;
92
93 critical_enter();
94 tm_apply_sdelta((tm_system_t *)&tm_current, &offset);
95 critical_exit();
96
97 tm_epochchange_handler_t * echandler = epoch_handlers_root;
98 while(echandler){
99 echandler -> func(&offset);
100 echandler = echandler->next;
101 }
102}

104uint16_t tm_sync_host_read_hook(ucdm_addr_t address){
105 if (tm_sync_sm.state == TM_SYNC_STATE_WAIT_DELAY_OUT){
106 tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_IN;
107 // Host read our timestamp for delay calculation.
108 // Store the timestamp sent out
109 tm_current_time(&(tm_sync_sm.t2));
110 }
111 return 0;

120void tm_sync_handler(ucdm_addr_t addr){
121 switch(tm_sync_sm.state){
122 case TM_SYNC_STATE_PREINIT:
123 break;
124 case TM_SYNC_STATE_IDLE:
125 if (ucdm_register[addr].data){
126 // Got sync with timestamp from the host.
127 tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_OUT;
128 tm_redirect_next_host_ts(&tm_sync_sm.t2p);
129 } else {
130 // Got sync without timestamp from the host.
131 tm_sync_sm.state = TM_SYNC_STATE_WAIT_FOLLOW_UP;
132 }
133 tm_current_time(&(tm_sync_sm.t1p));
134 break;
135 case TM_SYNC_STATE_WAIT_FOLLOW_UP:
136 // Got the sync timestamp from the host.
137 tm_sync_sm.state = TM_SYNC_STATE_WAIT_DELAY_OUT;
138 tm_redirect_next_host_ts(&tm_sync_sm.t2p);
139 break;
140 case TM_SYNC_STATE_WAIT_DELAY_OUT:
141 // This is handled in the register read function for the
142 // system time register of UCDM. t2 will be set there.
143 break;
144 case TM_SYNC_STATE_WAIT_DELAY_IN:
145 // Host returned its timestamp for delay calculation.
146 tm_redirect_next_host_ts(&tm_sync_sm.t1);
147 tm_sync_sm.state = TM_SYNC_STATE_IDLE;
148 // All information is now available. Calculate and apply.
149 tm_sync_apply();
150 ucdm_exception_status &= ~UCDM_EXST_TIMESYNC_REQ;
151 // TODO Schedule update of RTC in main loop from here
152 break;
153 default:
154 break;
155 }
156 return;
157}
