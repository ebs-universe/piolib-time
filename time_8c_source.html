206 void tm_stime_from_rtime(tm_real_t* rtime, tm_system_t * stime){
207  int32_t days = rtime->date - 1;
208  uint16_t syear = 0;
209  int16_t years = 0;
210 
211  if (!use_epoch){
212  return;
213  }
214 
215  days += days_to_month[rtime->month];
216  syear = gyear(&tm_internal_epoch);
217  years = (int16_t)(gyear(rtime) - syear);
218  if (rtime->month < 3){
219  years --;
220  } else {
221  uint16_t lyear = gyear(rtime);
222  if ((!(lyear % 4) && (lyear % 100)) || !(lyear % 400)){
223  days ++;
224  }
225  }
226 
227  days += (years * 365);
228 
229  while (years > 0){
230  if ((!(syear % 4) && (syear % 100)) || !(syear % 400)){
231  days ++;
232  }
233  syear ++;
234  years --;
235  }
236  while (years < 0){
237  if ((!(syear % 4) && (syear % 100)) || !(syear % 400)){
238  days --;
239  }
240  syear --;
241  years ++;
242  }
243 
244  *stime = (( days * TIME_SECONDS_PER_DAY * 1000LL) +
245  ( rtime->hours * TIME_SECONDS_PER_HOUR * 1000LL) +
246  (rtime->minutes * TIME_SECONDS_PER_MINUTE * 1000LL) +
247  (rtime->seconds * 1000LL) + rtime->millis +
248  tm_leapseconds - tm_internal_epoch_offset);
249 }

252 void tm_rtime_from_stime(tm_system_t* stime, tm_real_t* rtime){
253  if (!use_epoch){
254  return;
255  }
256 
257  *rtime = tm_internal_epoch;
258  tm_system_t stime_internal = *stime + tm_internal_epoch_offset - tm_leapseconds;
259  uint32_t days = stime_internal / (TIME_SECONDS_PER_DAY * 1000);
260  uint32_t remaining_ms = stime_internal % (TIME_SECONDS_PER_DAY * 1000);
261 
262  rtime->hours = remaining_ms / (TIME_SECONDS_PER_HOUR * 1000);
263  remaining_ms %= (TIME_SECONDS_PER_HOUR * 1000);
264  rtime->minutes = remaining_ms / (TIME_SECONDS_PER_MINUTE * 1000);
265  remaining_ms %= (TIME_SECONDS_PER_MINUTE * 1000);
266  rtime->seconds = remaining_ms / 1000;
267  rtime->millis = remaining_ms % 1000;
268 
269  uint16_t year = gyear(&tm_internal_epoch);
270  while (days >= 365) {
271 
272  uint16_t leap_adjustment = ((!((year+1)% 4) && ((year+1) % 100)) || !((year+1) % 400)) ? 366 : 365;
273  if (days < leap_adjustment) break;
274  days -= leap_adjustment;
275  year++;
276  }
277 
278  // Calculate month, starting from march
279  uint8_t next_month = 3;
280  uint8_t next_month_days = 31;
281  while (days >= next_month_days) {
282  days -= next_month_days;
283  if (next_month == 12) {
284  year ++;
285  next_month = 1;
286  } else {
287  next_month ++;
288  }
289  rtime->month = next_month;
290 
291  if (next_month == 2 && ((!(year % 4) && (year % 100)) || !(year % 400))){
292  next_month_days = 29;
293  } else {
294  next_month_days = days_in_month[next_month];
295  }
296  }
297  rtime->date = days + 1; // Days are 1-indexed
298  rtime->year = year % 100;
299  rtime->century = year / 100;
300 }

302 /*
303  * The internal epoch is used to allow arbitrary epoch setting.
304  * All our calculations related to rtime conversions are based on
305  * an epoch at midnight, the 1st of March. When another epoch is
306  * set, we choose a point just before the chosen epoch which has
307  * the needed characteristics as the internal epoch. We also
308  * maintain the offset between the actual epoch and the internal
309  * epoch.
310  *
311  * Timestamps in the system are all stored against the
312  * configured epoch. The internal epoch is only used ephemerally
313  * while converting between the tm_system_t and tm_real_t.
314  */

317 static void tm_set_internal_epoch(tm_real_t* rtime){
318  tm_internal_epoch.millis = 0;
319  tm_internal_epoch.seconds = 0;
320  tm_internal_epoch.minutes = 0;
321  tm_internal_epoch.hours = 0;
322  tm_internal_epoch.date = 1;
323  tm_internal_epoch.month = 3;
324  if (rtime->month >= 3){
325  tm_internal_epoch.year = rtime->year;
326  tm_internal_epoch.century = rtime->century;
327  }
328  else{
329  if (rtime->year){
330  tm_internal_epoch.year = rtime->year - 1;
331  tm_internal_epoch.century = rtime->century;
332  }
333  else{
334  tm_internal_epoch.year = 99;
335  tm_internal_epoch.century = rtime->century - 1;
336  }
337  }
338  uint16_t days = rtime->date - 1;
339  days += days_to_month[rtime->month];
340  tm_internal_epoch_offset = (( days * TIME_SECONDS_PER_DAY ) +
341  ( rtime->hours * TIME_SECONDS_PER_HOUR ) +
342  (rtime->minutes * TIME_SECONDS_PER_MINUTE ) +
343  (rtime->seconds )) * 1000 ;
344  return;
345 }

347 void tm_set_epoch(tm_real_t* rtime, uint8_t follow){
348  tm_sdelta_t sdelta;
349  tm_system_t original_epoch;
350  tm_system_t new_epoch;
351 
352 
353  if (follow){
354  tm_stime_from_rtime(&tm_epoch, &original_epoch);
355  tm_stime_from_rtime(rtime, &new_epoch);
356  tm_get_sdelta(&original_epoch, &new_epoch, &sdelta);
357  critical_enter();
358  tm_apply_sdelta((tm_system_t *)&tm_current, &sdelta);
359  }
360  else{
361  tm_clear_sdelta(&sdelta);
362  critical_enter();
363  tm_clear_stime((tm_system_t *)&tm_current);
364  tm_current = tm_internal_epoch_offset;
365  }
366 
367  memcpy((void*)(&tm_epoch), (void*)rtime, sizeof(tm_real_t));
368  tm_set_internal_epoch(rtime);
369 
370  use_epoch = 1;
371 
372  tm_epochchange_handler_t * echandler = epoch_handlers_root;
373  while(echandler){
374  echandler -> func(&sdelta);
375  echandler = echandler->next;
376  }
377 
378  critical_exit();
379  return;
380 }
