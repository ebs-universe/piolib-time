134uint8_t tm_check_invalid_rtime(tm_real_t* rtime){
135 if (rtime->century > 24) {
136 return 1;
137 }
138 if (rtime->year > 99) {
139 return 2;
140 }
141 if (rtime->month < 1 || rtime->month > 12) {
142 return 3;
143 }
144 if (rtime->date < 1 || rtime->date > 31) {
145 // This should actually check based on month
146 // and year
147 return 4;
148 }
149 if (rtime->hours > 23) {
150 return 5;
151 }
152 if (rtime->minutes > 59) {
153 return 6;
154 }
155 if (rtime->seconds > 59) {
156 return 7;
157 }
158 if (rtime->millis > 999) {
159 return 8;
160 }
161 return 0;

165void tm_sdelta_from_rdelta(tm_rdelta_t* rdelta, tm_sdelta_t* sdelta){
166 uint64_t result;
167 result = rdelta->days * TIME_SECONDS_PER_DAY * 1000LL;
168 result += rdelta->hours * TIME_SECONDS_PER_HOUR * 1000LL;
169 result += rdelta->minutes * TIME_SECONDS_PER_MINUTE * 1000LL;
170 result += rdelta->seconds * 1000LL;
171 result += rdelta->millis;
172 if (rdelta->sgn) {
173 result = result * -1;
174 }
175 *sdelta = result;
176}

178void tm_rdelta_from_sdelta(tm_sdelta_t* sdelta, tm_rdelta_t* rdelta){
179 tm_clear_rdelta(rdelta);
180 tm_sdelta_t sdelta_copy = *sdelta;
181
182 if (sdelta_copy < 0){
183 rdelta->sgn = 1;
184 sdelta_copy *= -1;
185 }
186
187 rdelta->days = sdelta_copy / (TIME_SECONDS_PER_DAY * 1000LL);
188 sdelta_copy -= rdelta->days * (TIME_SECONDS_PER_DAY * 1000LL);
189
190 rdelta->hours = sdelta_copy / (TIME_SECONDS_PER_HOUR * 1000LL);
191 sdelta_copy -= rdelta->hours * (TIME_SECONDS_PER_HOUR * 1000LL);
192
193 rdelta->minutes = sdelta_copy / (TIME_SECONDS_PER_MINUTE * 1000LL);
194 sdelta_copy -= rdelta->minutes * (TIME_SECONDS_PER_MINUTE * 1000LL);
195
196 rdelta->seconds = sdelta_copy / 1000LL;
197 sdelta_copy -= rdelta->seconds * 1000LL;
198
199 rdelta->millis = sdelta_copy;
200 return;
201}

216void tm_stime_from_rtime(tm_real_t* rtime, tm_system_t * stime){
217 int32_t days = rtime->date - 1;
218 uint16_t syear = 0;
219 int16_t years = 0;
220
221 if (!use_epoch){
222 return;
223 }
224
225 days += days_to_month[rtime->month];
226 syear = gyear(&tm_internal_epoch);
227 years = (int16_t)(gyear(rtime) - syear);
228 if (rtime->month < 3){
229 years --;
230 } else {
231 uint16_t lyear = gyear(rtime);
232 if ((!(lyear % 4) && (lyear % 100)) || !(lyear % 400)){
233 days ++;
234 }
235 }
236
237 days += (years * 365);
238
239 while (years > 0){
240 if ((!(syear % 4) && (syear % 100)) || !(syear % 400)){
241 days ++;
242 }
243 syear ++;
244 years --;
245 }
246 while (years < 0){
247 if ((!(syear % 4) && (syear % 100)) || !(syear % 400)){
248 days --;
249 }
250 syear --;
251 years ++;
252 }
253
254 *stime = (( days * TIME_SECONDS_PER_DAY * 1000LL) +
255 ( rtime->hours * TIME_SECONDS_PER_HOUR * 1000LL) +
256 (rtime->minutes * TIME_SECONDS_PER_MINUTE * 1000LL) +
257 (rtime->seconds * 1000LL) + rtime->millis +
258 tm_leapseconds - tm_internal_epoch_offset);
259}

262void tm_rtime_from_stime(tm_system_t* stime, tm_real_t* rtime){
263 if (!use_epoch){
264 return;
265 }
266
267 *rtime = tm_internal_epoch;
268 tm_system_t stime_internal = *stime + tm_internal_epoch_offset - tm_leapseconds;
269 uint32_t days = stime_internal / (TIME_SECONDS_PER_DAY * 1000);
270 uint32_t remaining_ms = stime_internal % (TIME_SECONDS_PER_DAY * 1000);
271
272 rtime->hours = remaining_ms / (TIME_SECONDS_PER_HOUR * 1000);
273 remaining_ms %= (TIME_SECONDS_PER_HOUR * 1000);
274 rtime->minutes = remaining_ms / (TIME_SECONDS_PER_MINUTE * 1000);
275 remaining_ms %= (TIME_SECONDS_PER_MINUTE * 1000);
276 rtime->seconds = remaining_ms / 1000;
277 rtime->millis = remaining_ms % 1000;
278
279 uint16_t year = gyear(&tm_internal_epoch);
280 while (days >= 365) {
281
282 uint16_t leap_adjustment = ((!((year+1)% 4) && ((year+1) % 100)) || !((year+1) % 400)) ? 366 : 365;
283 if (days < leap_adjustment) break;
284 days -= leap_adjustment;
285 year++;
286 }
287
288 // Calculate month, starting from march
289 uint8_t next_month = 3;
290 uint8_t next_month_days = 31;
291 while (days >= next_month_days) {
292 days -= next_month_days;
293 if (next_month == 12) {
294 year ++;
295 next_month = 1;
296 } else {
297 next_month ++;
298 }
299 rtime->month = next_month;
300
301 if (next_month == 2 && ((!(year % 4) && (year % 100)) || !(year % 400))){
302 next_month_days = 29;
303 } else {
304 next_month_days = days_in_month[next_month];
305 }
306 }
307 rtime->date = days + 1; // Days are 1-indexed
308 rtime->year = year % 100;
309 rtime->century = year / 100;
310}

312/*
313 * The internal epoch is used to allow arbitrary epoch setting.
314 * All our calculations related to rtime conversions are based on
315 * an epoch at midnight, the 1st of March. When another epoch is
316 * set, we choose a point just before the chosen epoch which has
317 * the needed characteristics as the internal epoch. We also
318 * maintain the offset between the actual epoch and the internal
319 * epoch.
320 *
321 * Timestamps in the system are all stored against the
322 * configured epoch. The internal epoch is only used ephemerally
323 * while converting between the tm_system_t and tm_real_t.
324 */
325static void tm_set_internal_epoch(tm_real_t* rtime);
326
327static void tm_set_internal_epoch(tm_real_t* rtime){
328 tm_internal_epoch.millis = 0;
329 tm_internal_epoch.seconds = 0;
330 tm_internal_epoch.minutes = 0;
331 tm_internal_epoch.hours = 0;
332 tm_internal_epoch.date = 1;
333 tm_internal_epoch.month = 3;
334 if (rtime->month >= 3){
335 tm_internal_epoch.year = rtime->year;
336 tm_internal_epoch.century = rtime->century;
337 }
338 else{
339 if (rtime->year){
340 tm_internal_epoch.year = rtime->year - 1;
341 tm_internal_epoch.century = rtime->century;
342 }
343 else{
344 tm_internal_epoch.year = 99;
345 tm_internal_epoch.century = rtime->century - 1;
346 }
347 }
348 uint16_t days = rtime->date - 1;
349 days += days_to_month[rtime->month];
350 tm_internal_epoch_offset = (( days * TIME_SECONDS_PER_DAY ) +
351 ( rtime->hours * TIME_SECONDS_PER_HOUR ) +
352 (rtime->minutes * TIME_SECONDS_PER_MINUTE ) +
353 (rtime->seconds )) * 1000 ;
354 return;
355}

357void tm_set_epoch(tm_real_t* rtime, uint8_t follow){
358 tm_sdelta_t sdelta;
359 tm_system_t original_epoch;
360 tm_system_t new_epoch;
361
362
363 if (follow){
364 tm_stime_from_rtime(&tm_epoch, &original_epoch);
365 tm_stime_from_rtime(rtime, &new_epoch);
366 tm_get_sdelta(&original_epoch, &new_epoch, &sdelta);
367 critical_enter();
368 tm_apply_sdelta((tm_system_t *)&tm_current, &sdelta);
369 }
370 else{
371 tm_clear_sdelta(&sdelta);
372 critical_enter();
373 tm_clear_stime((tm_system_t *)&tm_current);
374 tm_current = tm_internal_epoch_offset;
375 }
376
377 memcpy((void*)(&tm_epoch), (void*)rtime, sizeof(tm_real_t));
378 tm_set_internal_epoch(rtime);
379
380 use_epoch = 1;
381
382 tm_epochchange_handler_t * echandler = epoch_handlers_root;
383 while(echandler){
384 echandler -> func(&sdelta);
385 echandler = echandler->next;
386 }
387
388 critical_exit();
389 return;
390}
