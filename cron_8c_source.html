76void tm_cron_insert_job(cron_job_t * job_p){
77 cron_job_t * walker = cron_nextjob_p;
78 critical_enter();
79 job_p->active = 1;
80 if(walker == NULL){
81 cron_nextjob_p = job_p;
82 job_p->prevjob = NULL;
83 job_p->nextjob = NULL;
84 return;
85 }
86 while(walker){
87 if (tm_cmp_stime(&(walker->texec), &(job_p->texec)) == 1){
88 job_p->nextjob = walker;
89 job_p->prevjob = walker->prevjob;
90 if(walker->prevjob){
91 walker->prevjob->nextjob = job_p;
92 }
93 else{
94 cron_nextjob_p = job_p;
95 }
96 walker->prevjob = job_p;
97 return;
98 }
99 if (!(walker->nextjob)){
100 walker->nextjob = job_p;
101 job_p->prevjob = walker;
102 job_p->nextjob = NULL;
103 return;
104 }
105 walker = walker->nextjob;
106 };
107 critical_exit();
108}

111void tm_cron_cancel_job(cron_job_t * job_p){
112 if (job_p->nextjob){
113 job_p->nextjob->prevjob = job_p->prevjob;
114 }
115 if (job_p->prevjob){

126void tm_cron_poll(void){
127 if (!cron_nextjob_p){
128 return;
129 }
130 tm_system_t current;
131 tm_current_time(&current);
132
133 if (tm_cmp_stime(&(cron_nextjob_p->texec), &current) <= 0){
134 if (cron_nextjob_p->handler) {
135 cron_nextjob_p->handler();
136 }
137 if (cron_nextjob_p->tafter_p){
138 tm_apply_sdelta(&(cron_nextjob_p->texec), cron_nextjob_p->tafter_p);
139 tm_cron_replace_job(cron_nextjob_p);
140 }
141 else{
142 cron_nextjob_p->active = 0;
143 cron_nextjob_p = cron_nextjob_p->nextjob;
144 cron_nextjob_p->prevjob = NULL;
145 }
146 }
147}
