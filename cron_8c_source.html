78 void tm_cron_insert_job(cron_job_t * job_p){
79  cron_job_t * walker = cron_nextjob_p;
80  job_p->active = 1;
81  if(walker == NULL){
82  cron_nextjob_p = job_p;
83  job_p->prevjob = NULL;
84  job_p->nextjob = NULL;
85  return;
86  }
87  while(walker){
88  if (tm_cmp_stime(&(walker->texec), &(job_p->texec)) == 1){
89  job_p->nextjob = walker;
90  job_p->prevjob = walker->prevjob;
91  if(walker->prevjob){
92  walker->prevjob->nextjob = job_p;
93  }
94  else{
95  cron_nextjob_p = job_p;
96  }
97  walker->prevjob = job_p;
98  return;
99  }
100  if (!(walker->nextjob)){
101  walker->nextjob = job_p;
102  job_p->prevjob = walker;
103  job_p->nextjob = NULL;
104  return;
105  }
106  walker = walker->nextjob;
107  };
108 }

126 void tm_cron_poll(void){
127  if (!cron_nextjob_p){
128  return;
129  }
130  tm_system_t current;
131  tm_current_time(&current);
132 
133  if (tm_cmp_stime(&(cron_nextjob_p->texec), &current) <= 0){
